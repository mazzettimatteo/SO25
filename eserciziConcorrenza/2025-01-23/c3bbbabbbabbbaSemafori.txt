IDEA: cont e firstCall 

semaphore SA=1
semaphore SB=0

process A:
    bool firstCall=true
    while true:
        SA.P()
        if(firstCall){
            firstCall=false
        }
        else{
            <print('A')>
        }
        SB.V()
        SB.V()
        SB.V()
process B:
    int cont=0
    while true
        SB.P()
        cont++
        <print('B')>
        if(cont==3){
            cont=0
            SA.V()
        }
