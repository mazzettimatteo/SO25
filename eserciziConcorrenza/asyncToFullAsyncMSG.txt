arecv(pid_t sender){
    while(true){
        if(db[sender].size()>0){
            break;
        }
        incoming=nbrecieve(ANY)
        if(incoming!=NULL){
            db[incoming.snd].enqueue(incoming.msg)
        }
    }
    return db[sender].dequeue()
}
